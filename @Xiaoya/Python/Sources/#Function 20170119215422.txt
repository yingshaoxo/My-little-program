def simple_math():
    import random
    num = random.randrange(4)
    a = random.randrange(100)
    b = random.randrange(100)
    while a == 0 or b == 0:
        a = random.randrange(100)
        b = random.randrange(100)
    if num == 0:
        return str(a) + ' + ' + str(b) + ' = ' + str(a+b)
    elif num == 1:
        return str(a) + ' - ' + str(b) + ' = ' + str(a-b)
    elif num == 2:
        return str(a) + ' × ' + str(b) + ' = ' + str(a*b)
    elif num == 3:
        result = a/b
        while int(result) != a/b:
            a = random.randrange(1000)
            b = random.randrange(100)
            while b == 0:
                b = random.randrange(100)
            result = a/b
        return str(a) + ' ÷ ' + str(b) + ' = ' + str(int(result))
